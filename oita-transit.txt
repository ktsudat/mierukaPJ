    <script>
        // DOMContentLoadedは「ページ全体のHTMLが読み込み終わったら」という意味です
        document.addEventListener("DOMContentLoaded", () => {
            
            // 'fade-in-up'というクラスを持つすべてのHTML要素を探してリスト(配列)にします
            const elements = document.querySelectorAll('.fade-in-up');
            
            // IntersectionObserverというブラウザの機能を使って、要素が画面に入ったかを監視します
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    // isIntersectingがtrueなら、その要素が画面内に入ったことを意味します
                    if (entry.isIntersecting) {
                        // 要素に'visible'というクラスを追加します。これでCSS側のアニメーションが発動します
                        entry.target.classList.add('visible');
                        // 一度表示されたら、その要素の監視をやめます(上にスクロールして戻った際に何度もアニメーションさせないため)
                        observer.unobserve(entry.target);
                    }
                });
            }, { threshold: 0.1 }); // threshold: 0.1 は「要素の10%が画面に入ったら発火する」という条件です

            // 見つけたすべての要素(elements)を一つずつ取り出し、監視対象に登録します
            elements.forEach(el => observer.observe(el));
        });
    </script>
